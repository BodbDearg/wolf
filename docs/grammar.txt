Module:
    Scope <EOF>

Scope:
    Stmnts

Stmnts:
    Stmnt
    Stmnt Stmnts

Stmnt:
    PrintStmnt
    VarDecl
    AssignExpr

VarDecl:
    var Identifier = AssignExpr

AssignExpr:
    BinaryExpr
    BinaryExpr = AssignExpr

PrintStmnt
    print ( StrLit )
    print ( BinaryExpr )

OrExpr:
	AndExpr
	AndExpr or OrExpr

AndExpr:
	BinaryExpr
	BinaryExpr and AndExpr

BinaryExpr:
	UnaryExpr
	UnaryExpr + BinaryExpr
	UnaryExpr - BinaryExpr
	UnaryExpr / BinaryExpr
	UnaryExpr * BinaryExpr

UnaryExpr:
	PrimaryExpr
	-PrimaryExpr
	+PrimaryExpr
	(BinaryExpr)

PrimaryExpr:
    IntLit
    Identifier
    ReadnumExpr

ReadnumExpr
    readnum ( )
