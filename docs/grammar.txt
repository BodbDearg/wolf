Module:
    Scope <EOF>

Scope:
    Exprs

Exprs:
    Expr
    Expr Exprs

Expr:
    PrintExpr
    AssignExpr

AssignExpr:
    BinaryExpr
    BinaryExpr = AssignExpr

PrintExpr
    print ( StringLit )
    print ( BinaryExpr )

BinaryExpr:
	UnaryExpr
	UnaryExpr + BinaryExpr
	UnaryExpr - BinaryExpr
	UnaryExpr / BinaryExpr
	UnaryExpr * BinaryExpr

UnaryExpr:
	IntLit
	-IntLit
	+IntLit
	(BinaryExpr)
