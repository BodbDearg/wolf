Module:
    Scope <EOF>

Scope:
    Stmnts

Stmnts:
    Stmnt
    Stmnt Stmnts

Stmnt:
    PrintStmnt
    VarDecl
    AssignExpr

PrintStmnt
    print ( AssignExpr )

VarDecl:
    var Identifier = AssignExpr

AssignExpr:
    OrExpr
    OrExpr = AssignExpr

OrExpr:
	AndExpr
	AndExpr or OrExpr

AndExpr:
	NotExpr
	NotExpr and AndExpr

NotExpr:
	AddSubExpr
	not NotExpr

AddSubExpr:
	MulDivExpr
	MulDivExpr + AddSubExpr
	MulDivExpr - AddSubExpr

MulDivExpr:
	UnaryExpr
	UnaryExpr * MulDivExpr
	UnaryExpr / MulDivExpr

UnaryExpr:
	PrimaryExpr
	-PrimaryExpr
	+PrimaryExpr
	(AssignExpr)

PrimaryExpr:
	IntLit
	BoolLit
	StrLit
    Identifier
    ReadnumExpr

ReadnumExpr
    readnum ( )
