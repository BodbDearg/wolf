    Scope <EOF>

Scope:
    Stmnts

Stmnts:
    Stmnt
    Stmnt Stmnts

Stmnt:
    PrintStmnt
    VarDecl
    AssignExpr

PrintStmnt
    print ( AssignExpr )

VarDecl:
    var Identifier = AssignExpr

AssignExpr:
    OrExpr
    OrExpr = AssignExpr

OrExpr:
	AndExpr
	AndExpr or OrExpr

AndExpr:
	NotExpr
	NotExpr and AndExpr

NotExpr:
	BinaryExpr
	not NotExpr

BinaryExpr:
	UnaryExpr
	UnaryExpr + BinaryExpr
	UnaryExpr - BinaryExpr
	UnaryExpr / BinaryExpr
	UnaryExpr * BinaryExpr

UnaryExpr:
	PrimaryExpr
	-PrimaryExpr
	+PrimaryExpr
	(AssignExpr)

PrimaryExpr:
	IntLit
	BoolLit
	StrLit
    Identifier
    ReadnumExpr

ReadnumExpr
    readnum ( )
